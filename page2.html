    public void enemyCharge()
{
    switch (currentState)
    {
        //Enemy starts at idle
        case State.Idle:
            Debug.Log("1");
            // Check if the distance between the enemy and the player's charge transform is less than startIdleDistance
            if (Vector3.Distance(transform.position, playerChargeTransform.position) < startIdleDistance)
            {
                //If the enemy is not frozen
                if (freezeRot == false)
                {
                    // Rotate the enemy to face the player's charge transform
                    transform.LookAt(playerChargeTransform);
                    chargeCoolDown += Time.deltaTime; //Increase charge cooldown
                }
                //If the charge cooldown reaches its maximum value
                if (chargeCoolDown >= chargeCooldownMax)
                {
                    // Set the enemy's state to Aggro
                    SetState(State.Aggro);
                    chargeCoolDown = 0; //Reset charge cooldown
                }
            }
            break;

        // is in aggro
        case State.Aggro:
            Debug.Log("2"); 
            // If the distance between the enemy and the player's charge transform is less than or equal to chargeDistance
            if (Vector3.Distance(transform.position, playerChargeTransform.position) <= chargeDistance)
            {
                freezeRot = true; //Freeze rotation of the enemy
                rbEnemyCharge.constraints = RigidbodyConstraints.FreezeRotationY; //Freeze rotation along the Y-axis
                Debug.Log("freeze rot"); // Log message for debugging
                // Move the enemy forward based on chargeSpeed and deltaTime
                transform.position += transform.forward * chargeSpeed * Time.deltaTime;
                freezeChargeTimeRemaining += Time.deltaTime; //Increase freeze charge time remaining
            }
            //If the freeze charge time reaches its maximum value
            if (freezeChargeTimeRemaining >= freezeChargeTimeMax)
            {
                //Set the enemy's state to Tired
                SetState(State.Tired);
                freezeChargeTimeRemaining = 0; // Reset freeze charge time
            }
            break;
    }
}
