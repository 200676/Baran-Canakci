    public void Drift(float _angle)
{
    if (Input.GetKey(KeyCode.LeftShift))
    {
        //Check if the local Z position of the wheel collider is less than 1.6
        if (_wheelcollider.transform.localPosition.z < 1.6f) 
        {
            //Get the sideways friction of the wheel collider
            WheelFrictionCurve sidewaysFriction = _wheelcollider.sidewaysFriction;

            // Increase the friction if it's less than 1.6
            if (sidewaysFriction.stiffness < 1.6f)
            {
                sidewaysFriction.stiffness *= 1.5f; 
                sidewaysFriction.extremumSlip = 0.61f; 
                sidewaysFriction.extremumValue = 0.32f; 
                sidewaysFriction.asymptoteSlip = 1.81f; 
                sidewaysFriction.asymptoteValue = 0.20f; 
            }
            //Assign the updated friction to the wheel collider
            _wheelcollider.sidewaysFriction = sidewaysFriction;
        }
    }
    else
    {
        //Reset the sideways friction if the left shift key is not pressed
        ResetSidewaysFriction();
    }
}

private void ResetSidewaysFriction()
{
    //Check if the local Z position of the wheel collider is less than 0
    if(_wheelcollider.transform.localPosition.z < 0)
    {
        //resets to default sideways friction values
        WheelFrictionCurve defaultSidewaysFriction = new WheelFrictionCurve();
        defaultSidewaysFriction.stiffness = 1.4f; 
        defaultSidewaysFriction.extremumSlip = 0.4f; 
        defaultSidewaysFriction.extremumValue = 1.0f; 
        defaultSidewaysFriction.asymptoteSlip = 0.5f; 
        defaultSidewaysFriction.asymptoteValue = 0.75f; 

        //Checks if the current friction is bigger than default
        if (_wheelcollider.sidewaysFriction.stiffness > defaultSidewaysFriction.stiffness)
        {
            _wheelcollider.sidewaysFriction = defaultSidewaysFriction;
        }
    }
}

    [SerializeField] private Rigidbody Target; //Reference to the Rigidbody component of the target object
[SerializeField] private float maxSpeed = 0.0f;
[SerializeField] private float minSpeedArrow;
[SerializeField] private float maxSpeedArrow; 

[Header ("UI")]
[SerializeField] private Text speedText; 
public RectTransform arrow; //Reference to the RectTransform component of the arrow
private float speed = 0.0f; //Current speed of the target object

private void Update()
{
    //Calculate the speed of the target object in kilometers per hour
    speed = Target.velocity.magnitude * 3.6f;

    //Update the speed text if it's not null
    if (speedText != null)
    {
        speedText.text = ((int)(speed) + "");
    }

    //Update the rotation of the arrow if it's not null
    if (arrow != null)
    {
        // Calculate the rotation angle of the arrow based on the current speed
        float arrowRotation = Mathf.Lerp(minSpeedArrow, maxSpeedArrow, speed / maxSpeed);
        // Apply the rotation to the arrow
        arrow.localEulerAngles = new Vector3(0, 0, arrowRotation);
    }
}
